02 จงเขียนโปรแกรม เพื่อคํานวณหาปริมาตรของรูปทรงกระบอก Volume = Area x H โดยมีข้อมูลเข้า จาก input คือ ค่าของรัศมีของวงกลม (r: Radius) ที่เป็๋นฐานของรูปทรงกระบอก และค่าความ สูงของรูปทรงกระบอก (h: Height) ตามลําดับ และแสดงผลลัพธ์ที่ result โดย วิธีการคํานวณ ปริมาตร = π * r * r * h และ ทศนิยม 2 ตำแหน่ง

* input ที่ได้เป็น string ต้องแปลงเป็น Float ก่อน

Height:

Answer: 

                $(()=>{
                    $("#inpH").blur(btnCalVolume);
                })
              
                function btnCalVolume() {
                      var inpR = $("#inpR");
                      var inpH = $("#inpH");
                      var r = parseFloat(inpR.val());
                      var h = parseFloat(inpH.val());
                      var v = Math.PI * r * r * h;
                      $("#txtResult02").text(v.toFixed(2));
                    
                };

03 จงเขียนโปรแกรม เพื่อรับคะแนน 3 ช่อง โดยละช่อง มีคะแนน 0-50 โดยคะแนนประกอบด้วย การบ้าน (hw) สอบกลางภาค (mid) สอบปลายภาค (fin) โดยคิดสัดส่วน ดังนี้
score = (hw(30%) + mid(30%) + fin(40%)) * 2
หลังจากนั้น ตัดเกรด 3 ระดับ ดังนี้

  • X - excellence หากคะแนนรวม มากกว่าเท่ากับ 85%
  • P - Pass หากคะแนนรวม มากกว่าเท่ากับ 60%
  • F - Fail หากคะแนนรวม น้อยกว่า 60%

HW Midterm Final
Score Grade

Answer: 

                $(()=>{
                    $("#btnGrade").click(btnGrade_click);
                })
                function btnGrade_click() {
                  var iHW = $("#iHW").val();
                  var iMid = $("#iMid").val();
                  var iFin = $("#iFin").val();
                  var vs = (iHW * 30 / 100 + iMid * 30 / 100 + iFin * 40 / 100) * 2;
                  if (vs >= 85) {
                    $("#txtScore").text(vs);
                    $("#txtGd").text("X - excellence");
                  }
                  if (vs >= 60) {
                    $("#txtScore").text(vs);
                    $("#txtGd").text("P - Pass");
                  }
                  if (vs < 60) {
                    $("#txtScore").text(vs);
                    $("#txtGd").text("F - Fail");
                  }
                };

                    function btnClick(obj) {
                        alert(obj.id)   
                        var A = parseFloat($('#inpA').val());
                        var B = parseFloat($('#inpB').val());

        
                        if (obj.id == 'btnAdd') {
                            ans = A + B;
                        } else if (obj.id == 'btnSub') {
                            ans = A - B;
                        } else if (obj.id == 'btnMul') {
                            ans = A * B;
                        } else if (obj.id == 'btnDiv') {
                            ans = A / B;
                        }
        
                        $('#txtResult').text(ans);
                }
