03 จงเขียนโปรแกรม เพื่อรับคะแนน 3 ช่อง โดยละช่อง มีคะแนน 0-50 โดยคะแนนประกอบด้วย การบ้าน (hw) สอบกลางภาค (mid) สอบปลายภาค (fin) โดยคิดสัดส่วน ดังนี้
score = (hw(30%) + mid(30%) + fin(40%)) * 2
หลังจากนั้น ตัดเกรด 3 ระดับ ดังนี้

  • X - excellence หากคะแนนรวม มากกว่าเท่ากับ 85%
  • P - Pass หากคะแนนรวม มากกว่าเท่ากับ 60%
  • F - Fail หากคะแนนรวม น้อยกว่า 60%

HW Midterm Final
Score Grade

Answer: 

                $(()=>{
                    $("#btnGrade").click(grade_cal);
                })
                function grade_cal() {
                  var iHW = $("#iHW").val();
                  var iMid = $("#iMid").val();
                  var iFin = $("#iFin").val();

                  var sum = ((iHW * 0.3) + (iMid * 0.3) + (iFin * 0.4)) * 2;
                  if (sum >= 85) {
                    $("#txtScore").text(sum);
                    $("#txtGd").text("X - excellence");
                  }
                  if (sum >= 60) {
                    $("#txtScore").text(sum);
                    $("#txtGd").text("P - Pass");
                  }
                  if (sum < 60) {
                    $("#txtScore").text(sum);
                    $("#txtGd").text("F - Fail");
                  }
                };

                    function cal_volume() {
                      var start = parseInt($("#inpStart").val());
                      var stop = parseInt($("#inpStop").val());
                      var result = $("#txtResult04");
                      var sum = 0;
                      for (var i = start; i <= stop; i++) {

                      }
                      result.text(sum);
