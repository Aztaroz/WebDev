03 จงเขียนโปรแกรม เพื่อรับคะแนน 3 ช่อง โดยละช่อง มีคะแนน 0-50 โดยคะแนนประกอบด้วย การบ้าน (hw) สอบกลางภาค (mid) สอบปลายภาค (fin) โดยคิดสัดส่วน ดังนี้
score = (hw(30%) + mid(30%) + fin(40%)) * 2
หลังจากนั้น ตัดเกรด 3 ระดับ ดังนี้

  • X - excellence หากคะแนนรวม มากกว่าเท่ากับ 85%
  • P - Pass หากคะแนนรวม มากกว่าเท่ากับ 60%
  • F - Fail หากคะแนนรวม น้อยกว่า 60%

HW Midterm Final
Score Grade

Answer: 

                function btnGrade_click(){
                    var hw = parseInt(document.getElementById('iHW').value)
                    var mid = parseInt(document.getElementById('iMid').value)
                    var fin = parseInt(document.getElementById('iFin').value)

                    var score = ((hw*0.3)+(mid*0.3)+(fin*0.4))*2
                    txtScore.textContent = score

                    if (score >= 150*0.85){
                        txtGd.textContent = "X"
                    }
                    else if (score >= 150*0.6){
                        txtGd.textContent = "P"
                    }
                    else{
                        txtGd.textContent = "F"
                    }
                }

                function btnSum(){
                    var start = parseInt(document.getElementById("inpStart").value)
                    var stop = parseInt(document.getElementById("inpStop").value)
                    var sum = 0
                    for (var i = start; i<= stop; i++){
                        result = i+sum

                }
